  METHOD s_get_range.

* DEFINITION SECTION -------------------------- //

    DATA: lt_tvarvc       TYPE STANDARD TABLE OF tvarvc,

      FROM tvarvc
      INTO TABLE lt_tvarvc[]
      WHERE name = im_name
